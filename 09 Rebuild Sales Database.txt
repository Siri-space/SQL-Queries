--		DROP AND REBUILD TABLES

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'product_groups') DROP TABLE product_groups
CREATE TABLE product_groups (
	product_group_id INT IDENTITY(1,1) PRIMARY KEY,
	product_group_desc VARCHAR(30) NOT NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'products') DROP TABLE products
CREATE TABLE products (
	product_id INT IDENTITY(1,1) PRIMARY KEY,
	product_name VARCHAR(50) NOT NULL,
	product_desc VARCHAR(MAX) NULL,
	product_group_id INT NOT NULL,
	supplier_id INT NOT NULL,
	current_retail_price NUMERIC(7,2) NOT NULL,
	current_supplier_price NUMERIC(7,2) NOT NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'suppliers') DROP TABLE suppliers
CREATE TABLE suppliers (
	supplier_id INT IDENTITY(1,1) PRIMARY KEY,
	supplier_name VARCHAR(50) NOT NULL,
	address_1 VARCHAR(50) NULL,
	address_2 VARCHAR(50) NULL,
	address_3 VARCHAR(50) NULL,
	address_4 VARCHAR(50) NULL,
	postcode VARCHAR(12)  NULL,
	country_code CHAR(2) NULL,
	contact_name VARCHAR(50) NULL,
	contact_number VARCHAR(15) NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'product_price_history') DROP TABLE product_price_history
CREATE TABLE product_price_history (
	product_id INT NOT NULL,
	start_date DATE NOT NULL,
	end_date DATE NULL,
	price NUMERIC(7,2) NOT NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'pay_details') DROP TABLE pay_details
CREATE TABLE pay_details (
	customer_id INT NOT NULL PRIMARY KEY,
	long_card_number CHAR(16) NOT NULL,
	expiry_date CHAR(5)  NOT NULL,
	ccv CHAR(3) NOT NULL,
	name_on_card VARCHAR(30) NOT NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'countries') DROP TABLE countries
CREATE TABLE countries (
	country_code CHAR(2) NOT NULL PRIMARY KEY,
	country_name VARCHAR(30) NOT NULL,
	currency_code CHAR(3) NOT NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'customers') DROP TABLE customers
CREATE TABLE customers (
	customer_id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	title VARCHAR(10) NOT NULL,
	forename VARCHAR(30) NOT NULL,
	surname VARCHAR(30) NOT NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'customer_addresses') DROP TABLE customer_addresses
CREATE TABLE customer_addresses (
	customer_address_id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	customer_id INT NOT NULL,
	is_default_address BIT NOT NULL DEFAULT(1),
	address_1 VARCHAR(50) NOT NULL,
	address_2 VARCHAR(50) NULL,
	address_3 VARCHAR(50) NULL,
	address_4 VARCHAR(50) NULL,
	postcode VARCHAR(15) NOT NULL,
	country_code CHAR(2) NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'users') DROP TABLE users
CREATE TABLE users (
	user_id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	forename VARCHAR(30) NULL,
	surname VARCHAR(30) NULL,
	system_username VARCHAR(15) NULL,
	auth_level_id INT NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'auth_levels') DROP TABLE auth_levels
CREATE TABLE auth_levels (
	auth_level_id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	auth_level VARCHAR(30) NOT NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'payment_methods') DROP TABLE payment_methods
CREATE TABLE payment_methods (
	payment_method_id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	payment_method_desc VARCHAR(30) NOT NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'deliveries') DROP TABLE deliveries
CREATE TABLE deliveries (
	delivery_id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	product_id INT NOT NULL,
	supplier_id INT NOT NULL,
	user_id INT NOT NULL,
	delivery_date DATETIME NOT NULL DEFAULT(GETDATE()),
	total_qty_delivered INT NOT NULL,
	usable_quantity INT NOT NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'orders') DROP TABLE orders
CREATE TABLE orders (
	order_id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	order_date DATETIME NOT NULL DEFAULT(GETDATE()),
	product_id INT NOT NULL,
	customer_id INT NOT NULL,
	user_id INT NOT NULL,
	quantity INT NOT NULL,
	payment_method_id INT NOT NULL
	)
GO

IF EXISTS(SELECT name FROM sys.objects WHERE name = 'supplier_product_price_history') DROP TABLE supplier_product_price_history
CREATE TABLE supplier_product_price_history (
	product_id INT NOT NULL,
	start_date DATE NOT NULL,
	end_date DATE NULL,
	price NUMERIC(7,2) NOT NULL
	)
GO

SELECT p.product_id
	,p.product_name
	,p.supplier_id
	,ISNULL(d.qty_delivered,0) as qty_delivered
	,ISNULL(s.qty_sold,0) as qty_sold
	,ISNULL(d.qty_delivered,0) - ISNULL(s.qty_sold,0) as in_stock
	,CASE WHEN ISNULL(d.qty_delivered,0) - ISNULL(s.qty_sold,0) <= 10 THEN 'NEED TO RE-ORDER' ELSE '' END AS order_flag
	,(SELECT MAX(customer_id) as max_cust FROM orders) as h_ord
	,(SELECT MAX(customer_id) as max_cust FROM customers) as h_cust
	,(SELECT SUM(quantity) as tot_qty_sold FROM orders) as tot_qty_sold
	,DATEDIFF(DAY,(SELECT MAX(order_date) FROM orders),GETDATE())  as order_d_diff
	,(select count(*) from orders) as tot_sales
FROM products p
LEFT JOIN (
	SELECT product_id,SUM(quantity) as qty_sold
	from orders
	GROUP BY product_id
	) s ON s.product_id = p.product_id
LEFT JOIN (
	SELECT product_id,SUM(usable_quantity) as qty_delivered
	from deliveries
	GROUP BY product_id
	) d ON d.product_id = p.product_id
